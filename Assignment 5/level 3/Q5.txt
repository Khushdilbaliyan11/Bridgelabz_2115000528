public class Main{

    public static boolean isPrime(int number) {
        if (number <= 1) {
            return false;
        }
        for (int i = 2; i <= Math.sqrt(number); i++) {
            if (number % i == 0) {
                return false;
            }
        }
        return true;
    }

    public static boolean isNeon(int number) {
        int square = number * number;
        int sumOfDigits = 0;
        while (square > 0) {
            sumOfDigits += square % 10;
            square /= 10;
        }
        return sumOfDigits == number;
    }

    public static boolean isSpy(int number) {
        int sumOfDigits = 0;
        int productOfDigits = 1;
        while (number > 0) {
            int digit = number % 10;
            sumOfDigits += digit;
            productOfDigits *= digit;
            number /= 10;
        }
        return sumOfDigits == productOfDigits;
    }

    public static boolean isAutomorphic(int number) {
        int square = number * number;
        return String.valueOf(square).endsWith(String.valueOf(number));
    }

    public static boolean isBuzz(int number) {
        return number % 7 == 0 || String.valueOf(number).endsWith("7");
    }

    public static void main(String[] args) {
        int number = 25;

        System.out.println("Is " + number + " Prime? " + (isPrime(number) ? "Yes" : "No"));
        System.out.println("Is " + number + " Neon? " + (isNeon(number) ? "Yes" : "No"));
        System.out.println("Is " + number + " Spy? " + (isSpy(number) ? "Yes" : "No"));
        System.out.println("Is " + number + " Automorphic? " + (isAutomorphic(number) ? "Yes" : "No"));
        System.out.println("Is " + number + " Buzz? " + (isBuzz(number) ? "Yes" : "No"));
    }
}
