import java.util.Random;

public class Main{

    public static double[][] getEmployeeDetails(int numEmployees) {
        Random random = new Random();
        double[][] employeeDetails = new double[numEmployees][2];
        for (int i = 0; i < numEmployees; i++) {
            double salary = 10000 + random.nextInt(90000);
            int yearsOfService = 1 + random.nextInt(15);
            employeeDetails[i][0] = salary;
            employeeDetails[i][1] = yearsOfService;
        }
        return employeeDetails;
    }

    public static double[][] calculateNewSalariesAndBonuses(double[][] employeeDetails) {
        double[][] updatedDetails = new double[employeeDetails.length][3];
        for (int i = 0; i < employeeDetails.length; i++) {
            double oldSalary = employeeDetails[i][0];
            int yearsOfService = (int) employeeDetails[i][1];
            double bonus = 0;

            if (yearsOfService > 5) {
                bonus = 0.05 * oldSalary;
            } else {
                bonus = 0.02 * oldSalary;
            }

            double newSalary = oldSalary + bonus;
            updatedDetails[i][0] = oldSalary;
            updatedDetails[i][1] = newSalary;
            updatedDetails[i][2] = bonus;
        }
        return updatedDetails;
    }

    public static void calculateAndDisplayTotalAmounts(double[][] updatedDetails) {
        double totalOldSalary = 0, totalNewSalary = 0, totalBonus = 0;
        System.out.println("Employee Bonus and Salary Report");
        System.out.println("------------------------------------------------");
        System.out.printf("%-10s%-15s%-15s%-10s\n", "Employee", "Old Salary", "New Salary", "Bonus");
        System.out.println("------------------------------------------------");
        
        for (int i = 0; i < updatedDetails.length; i++) {
            double oldSalary = updatedDetails[i][0];
            double newSalary = updatedDetails[i][1];
            double bonus = updatedDetails[i][2];
            totalOldSalary += oldSalary;
            totalNewSalary += newSalary;
            totalBonus += bonus;

            System.out.printf("%-10d%-15.2f%-15.2f%-10.2f\n", i + 1, oldSalary, newSalary, bonus);
        }

        System.out.println("------------------------------------------------");
        System.out.printf("Total Old Salary: %.2f\n", totalOldSalary);
        System.out.printf("Total New Salary: %.2f\n", totalNewSalary);
        System.out.printf("Total Bonus Amount: %.2f\n", totalBonus);
    }

    public static void main(String[] args) {
        int numEmployees = 10;
        double[][] employeeDetails = getEmployeeDetails(numEmployees);
        double[][] updatedDetails = calculateNewSalariesAndBonuses(employeeDetails);
        calculateAndDisplayTotalAmounts(updatedDetails);
    }
}
